df['color'] = ['red' if x < 0.25 else 'orange' if x < 0.50 else 'lightgreen' if x < 0.75 else 'darkgreen' for x in df['ratings']/12.5]
df['sizes'] = [500 if y == True else 50 for y in df['has_guests']]

plt.figure(figsize=(30,10))
plt.scatter(x=df.episode_number, y=df.viewership_mil, c=df.color, s=df.sizes)

plt.title("Popularity, Quality, and Guest Appearances on the Office")
plt.xlabel("Episode Number")
plt.ylabel("Viewership in Millions")
